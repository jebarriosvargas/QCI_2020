@interact(t=(0,10,0.01))

    axes[0].set_xlim(-.4*ℓ  ,1.4*ℓ)
    axes[1].set_xlim(-.4*ℓ  ,1.4*ℓ)

    for nv in n:
        
        i = nv - 1
        E  = Eλ/ℓ/ℓ*nv*nv
        Ψe = exp(-1.J*E*t/ħ)*sqrt(2/ℓ)*sin(nv*π*x/ℓ)
        Ψe2 = conjugate(Ψe)*Ψe
