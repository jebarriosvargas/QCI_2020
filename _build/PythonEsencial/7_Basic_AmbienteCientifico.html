                          r"$\pi$",r"$2\pi$"])

axes[0].grid()

axes[1].set_title(r"${\rm Im}[e^{ix}]$")

axes[1].plot(x,z,'bo-',linewidth=3,markersize=5,markerfacecolor='red',
             markeredgecolor='green',label=r"$\sin(x)$")

axes[1].legend()

axes[1].set_xlabel(r"$x$")
